    class Bola:
    def __init__(self, warna):
        self.warna = warna
    def pantul(self):
        if self.warna == 'kuning':
            print('Bola kuning memantul tinggi!')
        else:
            print('Bola', self.warna, 'memantul biasa.')

b = Bola('kuning')
b.pantul()
